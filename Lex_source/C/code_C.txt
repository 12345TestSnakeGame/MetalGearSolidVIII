#include<stdio.h>
#define N 10

/* quitck sort */
void Qsort(int *a,int low ,int high)
{
    if(low>=high)
        return ;
    
	printf("start:%d  end:%d\n",low,high) ;
	int first = low ;
	int last  = high ;
	int key   = a[low] ;  
    while(first<last)
    {
        while(first<last && a[last]>=key)
            --last ;
        a[first] = a[last] ;
        while(first<last && a[first]<=key)
            ++first ;
        a[last] = a[first] ;
    }
    a[first] = key ;
    Qsort(a,low,first-1) ;
    Qsort(a,first+1,high) ;
}

int main(void)
{
    int a[N] = {5,3,8,6,0,9,1,7,4,2} ;
    int i;

    printf("original seq:") ;
    for(i=0;i<N;++i)
        printf("%d ",a[i]) ;
    printf("\n\n") ;

    /* quick sort */
    printf("range of each sort:\n") ;
    Qsort(a,0,N-1) ;

    printf("\n seq after sortï¼š") ;
    for(i=0;i<N;++i)
        printf("%d ",a[i]) ;
    printf("\n") ;
    return 0 ;
}
